    <script>

        function toggleStep(stepElement) {

            const isActive = stepElement.classList.contains('active');

           

            // Закрываем все шаги

            document.querySelectorAll('.step-container').forEach(step => {

                step.classList.remove('active');

            });

           

            // Открываем текущий шаг, если он не был активным

            if (!isActive) {

                stepElement.classList.add('active');

               

                // Анимируем прогресс-бары

                setTimeout(() => {

                    const progressBars = stepElement.querySelectorAll('.progress-fill');

                    progressBars.forEach(bar => {

                        const width = bar.style.width;

                        bar.style.width = '0%';

                        setTimeout(() => {

                            bar.style.width = width;

                        }, 100);

                    });

                }, 200);

            }

        }

 

        function switchTab(tabElement, contentId) {

            // Убираем активность со всех табов в этом контейнере

            const container = tabElement.closest('.step-content');

            container.querySelectorAll('.tab').forEach(tab => {

                tab.classList.remove('active');

            });

            container.querySelectorAll('.tab-content').forEach(content => {

                content.classList.remove('active');

            });

           

            // Активируем текущий таб и контент

            tabElement.classList.add('active');

            container.querySelector('#' + contentId).classList.add('active');

        }

 

        // Анимация появления при загрузке

        window.addEventListener('load', () => {

            const stepContainers = document.querySelectorAll('.step-container');

            stepContainers.forEach((container, index) => {

                setTimeout(() => {

                    container.style.opacity = '1';

                    container.style.transform = 'translateY(0)';

                }, index * 200);

            });

        });

 

        // Инициализация стилей для анимации

        document.querySelectorAll('.step-container').forEach(container => {

            container.style.opacity = '0';

            container.style.transform = 'translateY(20px)';

            container.style.transition = 'all 0.5s ease';

        });

 

        // Автопрокрутка к активному шагу

        function scrollToStep(stepElement) {

            stepElement.scrollIntoView({

                behavior: 'smooth',

                block: 'center'

            });

        }

 

        // Добавляем автопрокрутку при клике на шаг

        document.querySelectorAll('.step-container').forEach(step => {

            step.addEventListener('click', () => {

                setTimeout(() => scrollToStep(step), 300);

            });

        });
